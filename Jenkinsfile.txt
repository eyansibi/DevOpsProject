pipeline {
    agent any
    stages {
        stage('Checkout') {
            steps {
                git 'https://github.com/eyansibi/DevOpsProject.git'
            }
        }
        stage('Test') {
            steps {
                sh 'mvn test'
                junit 'target/surefire-reports/*.xml'
            }
        }
    }
    post {
        success {
            echo 'Tests réussis !'
        }
        failure {
            echo 'Échec des tests !'
        }
        always {
            archiveArtifacts artifacts: 'target/surefire-reports/*.xml', allowEmptyArchive: true
        }
    }
}
